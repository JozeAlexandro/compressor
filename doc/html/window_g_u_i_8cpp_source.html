114 void windowGUI::delegateExecRealWork( eTypeOfActions action )
115 {
117  if( !checkEndThreadFlag() )
118  return;
119 
121  if( ERR_STATUS_SUCCESS != mFileWorker.updateWriteFile("./temporary" ) )
122  {
123  mJouarnalModel.insertString( "Ошибка открытия файла!" );
124  }
125  else
126  {
127  mJouarnalModel.insertString( "Открыт файл для записи" );
128  mJouarnalModel.insertString( "Выполнение работы..." );
129 
131  mIsThreadEnd = false;
132 
133  std::thread thread( [ this, action ]( void )
134  {
136  auto [ newName, applyStatus ] =
137  mFileWorker.applyAlgorithm(
138  *mmAlgorithms.at( eTypeOfComprAlgorithm( mpUI->cbAlgorithmType->currentIndex() ) ),
139  action );
140 
142  mIsThreadEnd = true;
143  threadEnding( applyStatus, newName );
144  });
145 
147  thread.detach();
148 
150  mlblLoading.show();
151  }
152 }
